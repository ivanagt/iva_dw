🗂 Modelo de datos
-- Tres colecciones origen en MongoDB
evento_publico  →  18 tablas  (antejuicio, recede, memoria_electoral, ...)
atributos       →  17 tablas  (tse_2019_padron, gt_guatecompras, ...)
relaciones      →   2 tablas  (relaciones, relaciones_propietario)

-- Todas unidas con entidades por:

atributos.id_entidad_global                    →  entidades._id
relaciones.id_entidad_origen                   →  entidades._id
💡 Ejemplos SQL
Buscar persona en padrón 2023
SELECT nombre, dpi, departamento, municipio, centrovotacion
FROM v_tse_2023_padron
WHERE nombre ILIKE '%GARCIA LOPEZ%'
LIMIT 20;
Ver propiedades de una persona
SELECT nombre_origen, nombre_propiedad, tipo_propiedad,
       extension_terreno, valor_finca, registro_finca
FROM v_relaciones_propietario
WHERE dpi_origen = '1234567890101';
Ver sanciones de una entidad
SELECT nombre, lista, origen, tipo_lista, descripcion
FROM v_sanciones
WHERE nit = '12345678';
Cruzar RTU con padrón electoral
SELECT r.nombre, r.nit, r.actividad_principal,
       p.departamento, p.municipio
FROM v_gt_rtu_naturales r
JOIN v_tse_2023_padron p ON r.dpi = p.dpi
WHERE r.nombre ILIKE '%RODRIGUEZ%'
LIMIT 50;
Ver todas las vistas disponibles
SELECT table_name
FROM information_schema.views
WHERE table_schema = 'public'
ORDER BY table_name;
